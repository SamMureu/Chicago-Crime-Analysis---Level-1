WITH CrimeCounts AS ( SELECT primary_type, COUNT() AS total_crimes, SUM(CAST(arrest AS INT)) AS total_arrests, COUNT() - SUM(CAST(arrest AS INT)) AS total_non_arrests FROMbigquery-public-data.chicago_crime.crime WHERE EXTRACT(YEAR FROM date) = 2020 GROUP BY primary_type ORDER BY total_crimes DESC LIMIT 5 ) SELECT cc.primary_type, cc.total_crimes, cc.total_arrests, cc.total_non_arrests, ROUND((cc.total_arrests * 100.0) / NULLIF(cc.total_crimes, 0), 2) AS arrest_rate FROM CrimeCounts cc ORDER BY arrest_rate DESC;
